<objective>
## Goal
Move the maintenance form from a dialog into a dedicated full page at `/maintenance/new` and `/maintenance/:id/edit`, with field grouping via `FormSection`, a dirty-state blocker, and a loading/error skeleton on edit.

## Purpose
The maintenance form has 9 diverse fields across 4 concern groups (general info, cost/performer, schedule dates, notes) — it scrolls in the dialog and is cramped. Moving it to a full page gives space for clear section grouping, allows the user to navigate directly to an edit URL, and improves form discoverability.

## Output
- `src/pages/maintenance/ui/maintenance-form-page.tsx` — new full-page form (create + edit modes)
- `src/shared/config/routes.ts` — two new MAINTENANCE route constants
- `src/app/lib/router.tsx` — two new routes registered
- `src/pages/maintenance/ui/maintenance-page.tsx` — list page updated: "Thêm" → navigate, "Sửa" → navigate, dialog removed
- `src/pages/maintenance/index.ts` — exports updated
- `src/pages/maintenance/ui/maintenance-form-dialog.tsx` — deleted (no longer used)
</objective>

      const blocker = useBlocker(

      );

      function mapFetchError(error: unknown): string {

        if (e?.code === 'PGRST116' || e?.status === 406)
          return 'Không tìm thấy bản ghi bảo trì.';
        if (e?.status === 401 || e?.status === 403 || e?.code === 'PGRST301')
          return 'Phiên đăng nhập đã hết hạn. Vui lòng đăng nhập lại.';
        return 'Không thể tải bản ghi. Vui lòng thử lại.';
      }
